487 class rocfft_executor : public executor_base{
488 public:
490  using executor_base::forward;
492  using executor_base::backward;

496  template<typename index>
497  rocfft_executor(hipStream_t active_stream, box3d<index> const box, int dimension) :
498  stream(active_stream),
499  size(box.size[dimension]), size2(0),
500  howmanyffts(fft1d_get_howmany(box, dimension)),
501  stride(fft1d_get_stride(box, dimension)),
502  dist((dimension == box.order[0]) ? size : 1),
503  blocks((dimension == box.order[1]) ? box.osize(2) : 1),
504  block_stride(box.osize(0) * box.osize(1)),
505  total_size(box.count()),
506  embed({0, 0}),
507  worksize(compute_workspace_size())
508  {}
510  template<typename index>
511  rocfft_executor(hipStream_t active_stream, box3d<index> const box, int dir1, int dir2) :
512  stream(active_stream),
513  size(box.size[std::min(dir1, dir2)]), size2(box.size[std::max(dir1, dir2)]),
514  blocks(1), block_stride(0), total_size(box.count()),
515  worksize(0)
516  {
517  int odir1 = box.find_order(dir1);
518  int odir2 = box.find_order(dir2);
519 
520  if (std::min(odir1, odir2) == 0 and std::max(odir1, odir2) == 1){
521  stride = 1;
522  dist = size * size2;
523  embed = {static_cast<size_t>(stride), static_cast<size_t>(size)};
524  howmanyffts = box.size[2];
525  }else if (std::min(odir1, odir2) == 1 and std::max(odir1, odir2) == 2){
526  stride = box.size[0];
527  dist = 1;
528  embed = {static_cast<size_t>(stride), static_cast<size_t>(size) * static_cast<size_t>(stride)};
529  howmanyffts = box.size[0];
530  }else{ // case of directions (0, 2)
531  stride = 1;
532  dist = size;
533  embed = {static_cast<size_t>(stride), static_cast<size_t>(box.size[1]) * static_cast<size_t>(box.size[0])};
534  howmanyffts = box.size[1];
535  }
536  worksize = compute_workspace_size();
537  }
539  template<typename index>
540  rocfft_executor(hipStream_t active_stream, box3d<index> const box) :
541  stream(active_stream),
542  size(box.size[0]), size2(box.size[1]), howmanyffts(box.size[2]),
543  stride(0), dist(0),
544  blocks(1), block_stride(0),
545  total_size(box.count()),
546  embed({0, 0}),
547  worksize(compute_workspace_size())
548  {}

551  template<typename precision_type, direction dir>
552  void execute(std::complex<precision_type> data[], std::complex<precision_type> *workspace) const{
553  if (std::is_same<precision_type, float>::value){
554  if (dir == direction::forward)
555  make_plan(ccomplex_forward);
556  else
557  make_plan(ccomplex_backward);
558  }else{
559  if (dir == direction::forward)
560  make_plan(zcomplex_forward);
561  else
562  make_plan(zcomplex_backward);
563  }
564  rocfft_execution_info info;
565  rocfft_execution_info_create(&info);
566  rocfft_execution_info_set_stream(info, stream);
567 
568  size_t wsize = (std::is_same<precision_type, float>::value) ?
569  ((dir == direction::forward) ? ccomplex_forward->size_work() : ccomplex_backward->size_work()) :
570  ((dir == direction::forward) ? zcomplex_forward->size_work() : zcomplex_backward->size_work());
571 
572  if (wsize > 0)
573  rocfft_execution_info_set_work_buffer(info, reinterpret_cast<void*>(workspace), wsize);
574 
575  for(int i=0; i<blocks; i++){
576  void* block_data = reinterpret_cast<void*>(data + i * block_stride);
577  rocm::check_error( rocfft_execute(
578  (std::is_same<precision_type, float>::value) ?
579  ((dir == direction::forward) ? *ccomplex_forward : *ccomplex_backward) :
580  ((dir == direction::forward) ? *zcomplex_forward : *zcomplex_backward),
581  &block_data, nullptr, info), "rocfft execute");
582  }
583  rocfft_execution_info_destroy(info);
584  }

685  template<typename index>

702  template<typename precision_type>
703  void forward(precision_type const indata[], std::complex<precision_type> outdata[], std::complex<precision_type> *workspace) const{
704  if (std::is_same<precision_type, float>::value){
705  make_plan(sforward);
706  }else{
707  make_plan(dforward);
708  }
709 
710  rocfft_execution_info info;
711  rocfft_execution_info_create(&info);
712  rocfft_execution_info_set_stream(info, stream);
713 
714  size_t wsize = (std::is_same<precision_type, float>::value) ? sforward->size_work() : dforward->size_work();
715  if (wsize > 0)
716  rocfft_execution_info_set_work_buffer(info, reinterpret_cast<void*>(workspace), wsize);
717 
718  precision_type *copy_indata = reinterpret_cast<precision_type*>(
719  reinterpret_cast<unsigned char *>(workspace) + wsize);
720  backend::data_manipulator<tag::gpu>::copy_n(stream, indata, box_size(), copy_indata);
721 
722  for(int i=0; i<blocks; i++){
723  void *rdata = const_cast<void*>(reinterpret_cast<void const*>(copy_indata + i * rblock_stride));
724  void *cdata = reinterpret_cast<void*>(outdata + i * cblock_stride);
725  rocm::check_error( rocfft_execute(
726  (std::is_same<precision_type, float>::value) ? *sforward : *dforward,
727  &rdata, &cdata, info), "rocfft execute");
728  }
729  rocfft_execution_info_destroy(info);
730  }
732  template<typename precision_type>
733  void backward(std::complex<precision_type> indata[], precision_type outdata[], std::complex<precision_type> *workspace) const{
734  if (std::is_same<precision_type, float>::value){
735  make_plan(sbackward);
736  }else{
737  make_plan(dbackward);
738  }
739 
740  rocfft_execution_info info;
741  rocfft_execution_info_create(&info);
742  rocfft_execution_info_set_stream(info, stream);
743 
744  size_t wsize = (std::is_same<precision_type, float>::value) ? sbackward->size_work() : dbackward->size_work();
745  if (wsize > 0)
746  rocfft_execution_info_set_work_buffer(info, reinterpret_cast<void*>(workspace), wsize);
747 
748  std::complex<precision_type> *copy_indata = reinterpret_cast<std::complex<precision_type>*>(
749  reinterpret_cast<unsigned char *>(workspace) + wsize);
750  backend::data_manipulator<tag::gpu>::copy_n(stream, indata, complex_size(), copy_indata);
751 
752  for(int i=0; i<blocks; i++){
753  void *cdata = const_cast<void*>(reinterpret_cast<void const*>(copy_indata + i * cblock_stride));
754  void *rdata = reinterpret_cast<void*>(outdata + i * rblock_stride);
755  rocm::check_error( rocfft_execute(
756  (std::is_same<precision_type, float>::value) ? *sbackward : *dbackward,
757  &cdata, &rdata, info), "rocfft execute");
758  }
759  rocfft_execution_info_destroy(info);
760  }

763  forward<float>(indata, outdata, workspace);
764  }

772  }

775  backward<double>(indata, outdata, workspace);
776  }
777 
779  int box_size() const override{ return rsize; }
781  int complex_size() const override{ return csize; }
783  size_t workspace_size() const override{ return worksize; }
785  size_t compute_workspace_size() const{
786  make_plan(sforward);
787  make_plan(dforward);
788  make_plan(sbackward);
789  make_plan(dbackward);
790  // Temporary copies have to be made, request that from user in addition, to what rocFFT requires.
791  return
792  std::max( std::max(sforward->size_work() + box_size() * sizeof(float), sbackward->size_work() + complex_size() * sizeof(std::complex<float>)) / sizeof(std::complex<float>),
793  std::max(dforward->size_work() + box_size() * sizeof(double), dbackward->size_work() + complex_size() * sizeof(std::complex<double>)) / sizeof(std::complex<double>) ) + 1;
794  }
